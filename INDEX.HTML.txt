  <script>
    const entrada = document.getElementById("entrada");
    const boton = document.getElementById("buscar");
    const resultados = document.getElementById("resultados");

    boton.addEventListener("click", () => {
      const consulta = entrada.value.trim();
      resultados.innerHTML = "";

      if (consulta === "") {
        resultados.innerHTML = "<p>‚ö†Ô∏è Escrib√≠ algo para buscar.</p>";
        return;
      }

      resultados.innerHTML = "<p>üîé Buscando en Wikipedia...</p>";

      const url = `https://es.wikipedia.org/w/api.php?action=query&list=search&srsearch=${encodeURIComponent(consulta)}&format=json&origin=*`;

      fetch(url)
        .then(response => response.json())
        .then(data => {
          const resultadosBusqueda = data.query.search;
          if (resultadosBusqueda.length === 0) {
            resultados.innerHTML = "<p>‚ùå No se encontraron resultados.</p>";
            return;
          }

          resultados.innerHTML = "<h2>Resultados desde Wikipedia:</h2>";
          resultadosBusqueda.forEach(item => {
            const enlace = `https://es.wikipedia.org/?curid=${item.pageid}`;
            const div = document.createElement("div");
            div.className = "item";
            div.innerHTML = `
              <a href="${enlace}" target="_blank">${item.title}</a>
              <p>${item.snippet}...</p>
            `;
            resultados.appendChild(div);
          });
        })
        .catch(error => {
          console.error("Error al buscar:", error);
          resultados.innerHTML = "<p>‚ùå Error al conectar con Wikipedia.</p>";
        });
    });
  </script>
